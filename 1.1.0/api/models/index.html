    @classmethod
def from_api(cls, data: dict[str, Any]) -> "LatestResponse":
    """Create a LatestResponse from API data.

    The API returns {"data": {"item_id": {...}, ...}}.
    Item IDs are strings in the JSON but we convert to int.
    """
    prices = {
        int(item_id): LatestPrice.model_validate(price_data)
        for item_id, price_data in data.get("data", {}).items()
    }
    return cls(data=prices)

    @classmethod
def from_list(cls, data: list[dict[str, Any]]) -> "MappingResponse":
    """Create a MappingResponse from a list of item dicts.

    The API returns a raw list, not a wrapped object.
    """
    items = [ItemMapping.model_validate(item) for item in data]
    return cls(items=items)

    @classmethod
def from_api(
    cls, data: dict[str, Any], item_id: int | None = None
) -> "TimeseriesResponse":
    """Create a TimeseriesResponse from API data.

    The API returns {"data": [{...}, ...]}.

    Args:
        data: The raw API response data.
        item_id: The item ID this timeseries is for.
    """
    points = [
        TimeseriesDataPoint.model_validate(point) for point in data.get("data", [])
    ]
    return cls(item_id=item_id, data=points)
